```python
def _backup_current_target(self):
    """Backup the current target file"""
    target = self.target_var.get().strip()

    if not target or not os.path.isfile(target):
        # Try pattern if target is directory
        pattern = self.pattern_var.get().strip()
        if os.path.isdir(target) and pattern:
            target = os.path.join(target, pattern)

        if not os.path.isfile(target):
            messagebox.showwarning("No File", "No target file set. Target a file first.")
            return

    show_backup_dialog(self, target, self.engine, self._add_traceback)
```
