/prompt - Shows current system prompt (from config.yaml)

  /prompt <custom text> - Sets temporary prompt, REPLACES config prompt
  - Does NOT automatically inject {available_tools}
  - Model won't see tool schemas unless YOU manually include them in your
  custom prompt

  /prompt enable/disable - Toggles whether system prompt is used at all

  Tool schema injection:

  Tool schemas are controlled by /tools prompt enable/disable, which sets
  tools_prompt_enabled flag.

  If tools_prompt_enabled = True:
  - System prompt gets {available_tools} replaced with JSON schemas
  - Works with BOTH config.yaml prompt AND custom /prompt <text>

  So if you want tools in a custom prompt:
  /prompt You are a helpful assistant. Available tools: {available_tools}
  The {available_tools} placeholder will be replaced with the JSON schemas
  IF /tools prompt is enabled.

  TL;DR: /prompt sets the template, /tools prompt controls whether schemas
  get injected into that template.

  PHASE 1: INPUT ANALYSIS (Before Model Call)

  def prepare_system_prompt(user_input: str, history: ConversationHistory) 
  -> dict:
      """
      Analyze input and return system prompt configuration
      
      Returns:
          {
              'prompt_level': 'minimal'|'contextual'|'full',
              'prompt_text': str,
              'reasoning': str,
              'hooks_triggered': []
          }
      """

      hooks_triggered = []

      # ============ HOOK 1: CHAT DETECTION ============
      if _detect_pure_chat(user_input):
          hooks_triggered.append('CHAT_MODE')
          return {
              'prompt_level': 'minimal',
              'prompt_text': _get_minimal_prompt(),
              'reasoning': 'Pure chat detected - no tools needed',
              'hooks_triggered': hooks_triggered
          }

      # ============ HOOK 2: PROJECT STATE CHECK ============
      in_project = history.current_project_file is not None
      if in_project:
          hooks_triggered.append('IN_PROJECT')

      # ============ HOOK 3: TOOL USAGE HISTORY ============
      recent_tool_use = history.tools_used_in_last_n_turns(3)
      if recent_tool_use:
          hooks_triggered.append('RECENT_TOOLS')

      # ============ HOOK 4: CONTINUATION DETECTION ============
      if _is_continuation_phrase(user_input) and recent_tool_use:
          hooks_triggered.append('CONTINUATION')
          return {
              'prompt_level': 'full',
              'prompt_text': _get_full_prompt() +
  _get_project_context(history),
              'reasoning': 'Continuation of tool work detected',
              'hooks_triggered': hooks_triggered
          }

      # ============ HOOK 5: EXPLICIT TOOL REQUEST ============
      if _has_tool_keywords(user_input):
          hooks_triggered.append('EXPLICIT_TOOL_REQUEST')
          return {
              'prompt_level': 'full',
              'prompt_text': _get_full_prompt() +
  _get_project_context(history),
              'reasoning': 'Explicit tool/file reference detected',
              'hooks_triggered': hooks_triggered
          }

      # ============ HOOK 6: CONTEXTUAL QUESTION ============
      if _is_question(user_input) and in_project:
          hooks_triggered.append('PROJECT_QUESTION')
          return {
              'prompt_level': 'contextual',
              'prompt_text': _get_minimal_prompt() +
  _get_project_context(history),
              'reasoning': 'Question about current project - light context',
              'hooks_triggered': hooks_triggered
          }

      # ============ HOOK 7: DEFAULT FALLBACK ============
      if in_project:
          hooks_triggered.append('PROJECT_DEFAULT')
          return {
              'prompt_level': 'contextual',
              'prompt_text': _get_minimal_prompt() +
  _get_project_context(history),
              'reasoning': 'In project - maintain context',
              'hooks_triggered': hooks_triggered
          }
      else:
          hooks_triggered.append('IDLE_DEFAULT')
          return {
              'prompt_level': 'minimal',
              'prompt_text': _get_minimal_prompt(),
              'reasoning': 'Idle state - minimal prompt',
              'hooks_triggered': hooks_triggered
          }
