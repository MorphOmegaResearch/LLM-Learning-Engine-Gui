  def _calculate_chain_risk(tool_chain_json) -> str:
      """Calculate risk level for tool chain"""
      risk_scores = {
          'file_delete': 'HIGH',
          'bash_execute': 'MEDIUM',
          'file_write': 'LOW',
          'file_move': 'MEDIUM',
          'git_operations': 'MEDIUM'
      }

      max_risk = 'SAFE'
      for step in tool_chain_json.get("tools", []):
          step_risk = risk_scores.get(step.get("name"), 'SAFE')
          if step_risk == 'HIGH':
              max_risk = 'HIGH'
          elif step_risk == 'MEDIUM' and max_risk != 'HIGH':
              max_risk = 'MEDIUM'
          elif step_risk == 'LOW' and max_risk == 'SAFE':
              max_risk = 'LOW'

      return max_risk
