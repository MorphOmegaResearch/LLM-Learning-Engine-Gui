  async def execute(self, pattern: str, path: str = ".",
                   case_sensitive: bool = False, max_results: int = 100) -> 
  ToolResult:
      try:
          # Validate search path is within working directory
          safe_search_path = self._safe_path(path)
          if safe_search_path is None:
              return ToolResult(success=False, error="Access denied")

          if not safe_search_path.exists():
              return ToolResult(success=False, error=f"Path not found: 
  {path}")

          # Search for matching files
          results = []
          search_pattern = pattern if case_sensitive else pattern.lower()

          for file_path in safe_search_path.rglob('*'):
              # Validate each file is within working dir
              if not self._validate_path(str(file_path)):
                  continue

              file_name = file_path.name if case_sensitive else
  file_path.name.lower()

              # Pattern matching (glob or substring)
              if '*' in search_pattern or '?' in search_pattern:
                  from fnmatch import fnmatch
                  if fnmatch(file_name, search_pattern):

  results.append(str(file_path.relative_to(safe_search_path)))
              else:
                  if search_pattern in file_name:

  results.append(str(file_path.relative_to(safe_search_path)))

              if len(results) >= max_results:
                  break

          output = '\n'.join(results)
          return ToolResult(
              success=True,
              output=output,
              data={"matches": len(results), "truncated": len(results) >=
  max_results}
          )
      except Exception as e:
          return ToolResult(success=False, error=str(e))
