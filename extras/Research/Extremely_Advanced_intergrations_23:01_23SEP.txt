"""
DynamicQuantizationManager
- Picks a safe quantization tag for a given base model based on VRAM and preset.
- Ensures the model is available locally (one-time pull per tag per run).
- Minimizes churn via a cooldown window and per-session sticky choice.

Logs (grep-friendly):
  [Quant] Using model <tag>
  [Quant] Pulling model <tag> ...
  [Quant][WARN] <message>
"""

from __future__ import annotations
import os
import re
import shutil
import subprocess
import threading
import time
from dataclasses import dataclass
from typing import Dict, Optional, Tuple

# ---- helpers -----------------------------------------------------------------

def _which(cmd: str) -> str:
    return shutil.which(cmd) or cmd

def _run(cmd: list[str], timeout: int = 20) -> Tuple[int, str]:
    try:
        out = subprocess.check_output(cmd, text=True, timeout=timeout, stderr=subprocess.STDOUT)
        return 0, out
    except subprocess.CalledProcessError as e:
        return e.returncode, e.output or ""
    except Exception as e:
        return 1, str(e)

def _print_once(cache: set[str], key: str, msg: str) -> None:
    if key not in cache:
        print(msg)
        cache.add(key)

# ---- VRAM detectors -----------------------------------------------------------

def _vram_nvidia_gb() -> float:
    code, out = _run(["nvidia-smi", "--query-gpu=memory.total", "--format=csv,noheader,nounits"], timeout=4)
    if code != 0:
        return -1.0
    vals = []
    for line in out.splitlines():
        line = line.strip()
        if line:
            try:
                # nvidia-smi returns MB units w/ nounits; normalize to GB
                mb = float(line)
                vals.append(mb / 1024.0)
            except Exception:
                pass
    return max(vals) if vals else -1.0

def _vram_amd_gb() -> float:
    # rocm-smi outputs vary; best-effort parse of "Total VRAM Memory (B): <bytes>"
    code, out = _run(["rocm-smi"], timeout=4)
    if code != 0:
        return -1.0
    # try to find something like "Total VRAM Memory (B): 17179869184"
    m = re.search(r"Total\s+VRAM\s+Memory\s+\(B\)\s*:\s*(\d+)", out, re.IGNORECASE)
    if not m:
        # fallback: look for "VRAM Total Memory (B)" variants
        m = re.search(r"VRAM.*?Total.*?\(B\)\s*:\s*(\d+)", out, re.IGNORECASE | re.DOTALL)
    if not m:
        return -1.0
    try:
        b = float(m.group(1))
        return b / (1024.0 ** 3)
    except Exception:
        return -1.0

def detect_vram_gb() -> float:
    """Return GPU VRAM (GB). If no GPU detected, returns 0.0 (CPU/low-resource path)."""
    v = _vram_nvidia_gb()
    if v >= 0:
        return v
    v = _vram_amd_gb()
    if v >= 0:
        return v
    return 0.0

def _inject_quant_tag(base_model: str, quant: str) -> str:
    # If base_model already has a colon, replace the trailing part after ':' with quant if it looks like a quant,
    # otherwise append ":<quant>".
    if ":" in base_model:
        # Some models include versions after colon; we append quant if there is not already a Q*-style suffix
        if re.search(r":Q[0-9].*", base_model):
            return re.sub(r":Q[0-9][^:]*$", f":{quant}", base_model)
        return f"{base_model}:{quant}"
    return f"{base_model}:{quant}"

class DynamicQuantizationManager:
    """
    Safe, sticky quant chooser with one-time model pulls.
    """
    def __init__(self, ollama_cmd: str = "ollama", pull_timeout_s: int = 1800,
                 cooldown_s: int = 30, dry_run: bool = False):
        self.ollama = _which(ollama_cmd)
        self.pull_timeout_s = pull_timeout_s
        self.cooldown_s = cooldown_s
        self.dry_run = dry_run
        self._lock = threading.Lock()
        self._pulled_once: set[str] = set()
        self._printed_once: set[str] = set()
        self._last_choice: Optional[Tuple[str, float]] = None  # (tag, timestamp)

    # --- discovery -------------------------------------------------------------

    def installed_models(self) -> set[str]:
        code, out = _run([self.ollama, "list"], timeout=6)
        if code != 0:
            return set()
        names = set()
        for line in out.splitlines():
            parts = line.split()
            if parts:
                names.add(parts[0])  # "name[:tag]"
        return names

    # --- policy ----------------------------------------------------------------

    def recommend_quantization(self, base_model: str,
                               resource_preset: str,
                               signals: Optional[Dict] = None) -> str:
        """
        Returns a model tag like "gemma-3-4b-it:Q5_K_M".
        - Uses VRAM and preset as primary signal.
        - Uses 'complexity' and 'resource_cost' to bump/degrade one step.
        - Respects a short cooldown to avoid rapid thrashing.
        """
        base = _normalize_family(base_model)
        preset = (resource_preset or _DEFAULT_PRESET).lower()
        vr_gb = detect_vram_gb()

        # start with default ladder for the preset
        ladder = list(_QUANT_TABLE.get(preset, _QUANT_TABLE[_DEFAULT_PRESET]))

        # coarse VRAM guardrails
        if vr_gb < 4.0 and "Q4_K_M" in ladder:
            ladder = ["Q4_K_M"] + [q for q in ladder if q != "Q4_K_M"]
        elif vr_gb >= 12.0 and "Q6_K" in ladder:
            ladder = ["Q6_K"] + [q for q in ladder if q != "Q6_K"]

        # adjust by signals
        s = signals or {}
        complexity = float(s.get("complexity", 0.5))
        resource_cost = float(s.get("resource_cost", 0.5))
        # prefer higher quant for high complexity when VRAM permits
        if complexity >= 0.75 and vr_gb >= 8.0 and "Q6_K" in ladder:
            if ladder[0] != "Q6_K":
                ladder.remove("Q6_K")
                ladder.insert(0, "Q6_K")
        # prefer lower quant for high resource cost / low VRAM
        if resource_cost >= 0.75 and "Q4_K_M" in ladder:
            if ladder[0] != "Q4_K_M":
                ladder.remove("Q4_K_M")
                ladder.insert(0, "Q4_K_M")

        # cooldown: if last tag still appropriate and cooldown not elapsed, keep it
        now = time.time()
        if self._last_choice and (now - self._last_choice[1] < self.cooldown_s):
            sticky_tag = self._last_choice[0]
            # keep if family matches and quant still in ladder (prevents oscillation)
            if sticky_tag.split(":")[0] == base and sticky_tag.split(":")[-1] in ladder:
                return sticky_tag

        tag = _inject_quant_tag(base, ladder[0])
        self._last_choice = (tag, now)
        return tag

    # --- provisioning ----------------------------------------------------------

    def ensure_model_available(self, model_tag: str) -> bool:
        """
        True when model is present locally (pull if needed, once per tag per run).
        Falls back down the ladder if pull fails.
        """
        base, quant = model_tag.split(":", 1) if ":" in model_tag else (model_tag, "")
        with self._lock:
            installed = self.installed_models()
            if model_tag in installed or base in installed:
                _print_once(self._printed_once, f"using:{model_tag}", f"[Quant] Using model {model_tag}")
                return True

            if self.dry_run:
                _print_once(self._printed_once, f"dry:{model_tag}", f"[Quant] (dry-run) would pull {model_tag}")
                return True

            # attempt pull; if fail, try degraded quants if applicable
            degrade = ["Q6_K", "Q5_K_M", "Q4_K_M"]
            if quant and quant in degrade:
                ladder = [quant] + [q for q in degrade if q != quant]
            else:
                ladder = degrade

            for q in ladder:
                cand = _inject_quant_tag(base, q)
                if cand in self._pulled_once:
                    _print_once(self._printed_once, f"using:{cand}", f"[Quant] Using model {cand}")
                    return True

                _print_once(self._printed_once, f"pull:{cand}", f"[Quant] Pulling model {cand} ...")
                code, out = _run([self.ollama, "pull", cand], timeout=self.pull_timeout_s)
                if code == 0:
                    self._pulled_once.add(cand)
                    print(f"[Quant] Using model {cand}")
                    return True
                else:
                    print(f"[Quant][WARN] Pull failed for {cand}: {out.strip()[:160]}")

            # final fallback: try base without quant
            _print_once(self._printed_once, f"pull:{base}", f"[Quant] Pulling model {base} ...")
            code, out = _run([self.ollama, "pull", base], timeout=self.pull_timeout_s)
            if code == 0:
                self._pulled_once.add(base)
                print(f"[Quant] Using model {base}")
                return True

            print(f"[Quant][WARN] Could not make model available: {model_tag}")
            return False

"""
MVCO: Markov-like lightweight context orchestrator
- Predicts a confidence score and optional hints for routing/model choice.
- Learns asynchronously from interaction outcomes.
- Guarantees clean shutdown (no zombie threads).

Logs (grep-friendly):
  [MVCO] conf=<0.00..1.00>
  [MVCO] shutdown OK
  [MVCO][WARN] <message>
"""

from __future__ import annotations
import json
import os
import queue
import threading
import time
from dataclasses import dataclass, asdict
from typing import Dict, Optional

# Optional persistence location (kept small; safe under home dir)
_DEFAULT_STATE_DIR = os.path.expanduser("~/.opencode/state")
_DEFAULT_STATE_FILE = os.path.join(_DEFAULT_STATE_DIR, "mvco_state.json")

@dataclass
class _MVCOState:
    # running confidence estimate (EWMA)
    conf: float = 0.5
    # basic outcome counts by model tag, lane, etc.
    total: int = 0
    ok: int = 0
    by_model_ok: Dict[str, int] = None
    by_model_total: Dict[str, int] = None

    def __post_init__(self):
        if self.by_model_ok is None:
            self.by_model_ok = {}
        if self.by_model_total is None:
            self.by_model_total = {}

class MVCO:
    """
    Minimal, safe learning scaffold.
    - predict(route_meta) -> dict(boost_temp, prefer_model, conf)
    - learn(outcome)      -> queue outcomes; learner thread updates state
    - shutdown()          -> join learner thread (prints [MVCO] shutdown OK)
    """
    def __init__(self, lr: float = 0.1, persist: bool = False, state_path: Optional[str] = None):
        self.lr = float(lr)
        self.persist = bool(persist)
        self.state_path = state_path or _DEFAULT_STATE_FILE
        self._q: "queue.Queue[dict]" = queue.Queue()
        self._stop = threading.Event()
        self._t = threading.Thread(target=self._loop, daemon=True)
        self._lock = threading.Lock()
        self._state = self._load_state()
        self._t.start()

    # ---- public API -----------------------------------------------------------

    def predict(self, route_meta: Dict) -> Dict:
        """
        Uses lightweight signals to return:
          - conf: current confidence [0.01..0.99]
          - prefer_model: None (placeholder hook for future policy)
          - boost_temp: 0.0 (placeholder; caller may adjust gen params externally)
        """
        with self._lock:
            conf = float(self._state.conf)
        # bounded
        conf = max(0.01, min(0.99, conf))
        print(f"[MVCO] conf={conf:.2f}")
        return {"conf": conf, "prefer_model": None, "boost_temp": 0.0}

    def learn(self, outcome: Dict) -> None:
        """
        Queue an outcome dict; examples:
          {"success": True, "model": "gemma-3-4b-it:Q5_K_M", "latency_ms": 820, "lane":"code"}
        """
        try:
            self._q.put_nowait(dict(outcome))
        except Exception:
            pass  # drop silently if queue is full; keep UI responsive

    def shutdown(self, timeout: float = 2.0) -> None:
        self._stop.set()
        try:
            self._t.join(timeout=timeout)
            print("[MVCO] shutdown OK")
        except Exception as e:
            print(f"[MVCO][WARN] shutdown error: {e}")

    # ---- worker ---------------------------------------------------------------

    def _loop(self):
        """Background learning loop."""
        last_persist = time.time()
        while not self._stop.is_set():
            try:
                outcome = self._q.get(timeout=0.5)
            except queue.Empty:
                # gentle idle persistence every ~30s
                if self.persist and (time.time() - last_persist) > 30:
                    self._save_state()
                    last_persist = time.time()
                continue

            ok = bool(outcome.get("success", True))
            model = str(outcome.get("model", "unknown"))
            lane = str(outcome.get("lane", "unknown"))  # reserved for future use

            with self._lock:
                # EWMA of success
                prev = self._state.conf
                target = 1.0 if ok else 0.0
                self._state.conf = 0.9 * prev + 0.1 * target  # gentle update
                # counts
                self._state.total += 1
                self._state.ok += 1 if ok else 0
                self._state.by_model_total[model] = self._state.by_model_total.get(model, 0) + 1
                if ok:
                    self._state.by_model_ok[model] = self._state.by_model_ok.get(model, 0) + 1

            # occasional persistence
            if self.persist and (time.time() - last_persist) > 10:
                self._save_state()
                last_persist = time.time()

        # flush on stop
        if self.persist:
            self._save_state()

    # ---- persistence ----------------------------------------------------------

    def _load_state(self) -> _MVCOState:
        if not self.persist:
            return _MVCOState()
        try:
            if os.path.isfile(self.state_path):
                with open(self.state_path, "r", encoding="utf-8") as f:
                    raw = json.load(f)
                st = _MVCOState(**raw)
                return st
        except Exception:
            pass
        return _MVCOState()

    def _save_state(self) -> None:
        if not self.persist:
            return
        try:
            os.makedirs(os.path.dirname(self.state_path), exist_ok=True)
            with open(self.state_path, "w", encoding="utf-8") as f:
                json.dump(asdict(self._state), f)
        except Exception:
            # never crash the app on persistence issues
            pass
