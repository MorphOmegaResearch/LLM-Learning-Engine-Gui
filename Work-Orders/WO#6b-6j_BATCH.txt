```python
# Data/tabs/training_tab/training_tab.py

from Data import config as C  # ensure present at top

def apply_plan(self, variant_id: str | None = None, training_profile: dict | None = None):
    try:
        tp = training_profile or (C.load_training_profile(variant_id) or {})
        if not tp:
            print("[TrainingTab] apply_plan: no profile")
            return

        # 1) Model Selection
        base = tp.get("base_model", "")
        if hasattr(self, "model_selection_panel") and hasattr(self.model_selection_panel, "set_base_model"):
            self.model_selection_panel.set_base_model(base)

        # 2) Scripts
        scripts = tp.get("selected_scripts", [])
        if hasattr(self, "runner_panel") and hasattr(self.runner_panel, "load_scripts"):
            self.runner_panel.load_scripts(scripts)

        # 3) Prompts & Schemas
        if hasattr(self, "load_prompts"):
            self.load_prompts(tp.get("selected_prompts", []))
        if hasattr(self, "load_schemas"):
            self.load_schemas(tp.get("selected_schemas", []))

        # 4) Runner settings
        if hasattr(self, "runner_panel") and hasattr(self.runner_panel, "load_settings"):
            self.runner_panel.load_settings(tp.get("runner_settings", {}))

        # 5) Profiles dropdown
        self.active_trainee_name = variant_id or tp.get("trainee_name")
        if hasattr(self, "profiles_panel") and hasattr(self.profiles_panel, "select_profile"):
            self.profiles_panel.select_profile(self.active_trainee_name)

        # 6) Refresh preview if present
        if hasattr(self, "update_preview"):
            self.update_preview()

        print(f"[TrainingTab] Plan applied for {self.active_trainee_name}")
    except Exception as e:
        print("[TrainingTab] Error in apply_plan:", e)
```

```python
# Data/config.py
import shutil, os
from pathlib import Path

def _mp_path(variant_id:str) -> Path:
    return Path(MODEL_PROFILES_DIR) / f"{variant_id}.json"

def _tp_path(variant_id:str) -> Path:
    return Path(TRAINING_PROFILES_DIR) / f"{variant_id}.json"

def rename_variant(old:str, new:str):
    # rename model & training profiles together
    for getter in (_mp_path, _tp_path):
        p_old, p_new = getter(old), getter(new)
        if p_old.exists():
            p_new.parent.mkdir(parents=True, exist_ok=True)
            os.replace(p_old, p_new)

def duplicate_variant(src:str, dst:str):
    for getter in (_mp_path, _tp_path):
        p_src, p_dst = getter(src), getter(dst)
        if p_src.exists():
            p_dst.parent.mkdir(parents=True, exist_ok=True)
            shutil.copy2(p_src, p_dst)

def delete_variant(variant_id:str):
    for getter in (_mp_path, _tp_path):
        p = getter(variant_id)
        if p.exists():
            p.unlink()
```

```python
def _collections_get_selected_variant(self):
    sel = self.collections_list.curselection()
    if not sel: return None
    it = (self._collections_cache or [])[sel[0]]
    if it.get("header"): return None
    return it.get("variant_id")

def _collections_rename(self):
    v = self._collections_get_selected_variant()
    if not v: return
    new = simpledialog.askstring("Rename Variant", f"New name for '{v}':", parent=self)
    if not new: return
    from Data import config as C
    C.rename_variant(v, new)
    self.refresh_collections_panel()

def _collections_duplicate(self):
    v = self._collections_get_selected_variant()
    if not v: return
    new = simpledialog.askstring("Duplicate Variant", f"Duplicate '{v}' to:", parent=self)
    if not new: return
    from Data import config as C
    C.duplicate_variant(v, new)
    self.refresh_collections_panel()

def _collections_delete(self):
    v = self._collections_get_selected_variant()
    if not v: return
    if messagebox.askyesno("Delete Variant", f"Delete '{v}' (model+training profile)?"):
        from Data import config as C
        C.delete_variant(v)
        self.refresh_collections_panel()
```

```python
def _prefill_variant_name(self, base_model:str, type_id:str):
```

```python
import json, os

def load_ui_state():

        return data.get("ui_state", {})
    except Exception:
        return {}

def save_ui_state(state:dict):
    try:
        data = json.load(open(SETTINGS_PATH, "r", encoding="utf-8")) if os.path.exists(SETTINGS_PATH) else {}
        data["ui_state"] = state
        json.dump(data, open(SETTINGS_PATH, "w", encoding="utf-8"), indent=2)

        print("[Settings] save_ui_state error:", e)
```
