```python
# Replace individual canvases with one master canvas
sidebar = ttk.Frame(main_split_left)  # wherever current left stack is
sidebar.grid(row=0, column=0, sticky=tk.NSEW)
sidebar.columnconfigure(0, weight=1); sidebar.rowconfigure(0, weight=1)

master_canvas = tk.Canvas(sidebar, bg="#2b2b2b", highlightthickness=0)
master_sb     = ttk.Scrollbar(sidebar, orient=tk.VERTICAL, command=master_canvas.yview)
stack_frame   = ttk.Frame(master_canvas)

master_canvas.create_window((0,0), window=stack_frame, anchor="nw")

master_canvas.grid(row=0, column=0, sticky=tk.NSEW)

def _resize(_e=None):

    master_canvas.itemconfig(1, width=master_canvas.winfo_width())

stack_frame.bind("<Configure>", _resize)
```

2. Move existing sections (Base/Available, Ollama, Collections) into `stack_frame` as:

```python
def make_section(parent, title, expanded_var: tk.BooleanVar):
    hdr = ttk.Frame(parent); hdr.pack(fill=tk.X, pady=(10,2))
    btn = ttk.Button(hdr, width=2, text="▼" if expanded_var.get() else "▶")
    btn.pack(side=tk.LEFT, padx=(0,6))
    ttk.Label(hdr, text=title, style='CategoryPanel.TLabel').pack(side=tk.LEFT)

    body = ttk.Frame(parent); 
    if expanded_var.get():
        body.pack(fill=tk.BOTH, expand=True)
    def _toggle():
        if expanded_var.get():
            # collapse
            for w in list(body.pack_slaves()):
                pass
            body.forget()
            expanded_var.set(False)
            btn.config(text="▶")
        else:
            body.pack(fill=tk.BOTH, expand=True)
            expanded_var.set(True)
            btn.config(text="▼")
        _resize()
    btn.config(command=_toggle)
    return hdr, body
```

```python
def _emit_json_event(self, widget, name: str, payload: dict):
```
